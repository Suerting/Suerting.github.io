                return num
            },
            set(value){
                console.log('有人修改age，反过来num也能修改,只要有人修改，就调用set')

# 事件修饰符

<div id="root">

     <!-- 阻止事件冒泡 ，点击按钮，冒泡到div元素,由内到外-->

# 滚轮

scroll是页面滚动条滚动会触发的事件，而wheel是鼠标滚轮滚动会触发的事件。 一旦滚动条到底部后，滑动鼠标滚轮继续滚动，wheel就会一直触发，而scroll不会触发。 scroll需等待回调函数执行完毕才触发，wheel无需等待回调函数执行完毕就可以触发

<div id="root">
    <ul @scroll="scroll" class="list1">
                <Li>1</Li>
                <Li>2</Li>
                <Li>3</Li>
                <Li>4</Li>
    </ul>
    <!-- 绑定滚动事件 @wheel滚动轮的滚动-->
    <ul @wheel.passive="wheel" class="list2">
                <Li>1</Li>
                <Li>2</Li>
                <Li>3</Li>
                <Li>4</Li>
    </ul>
<div id="root">

<script type="text/javascript">
     Vue.config.productionTip=false // 阻止vue在启动时生成生产提示
     const vm=new Vue({
          el:'#root',
          data:({
                
            }),
            methods:{
                // 事件的回调都配置在methods中，最终会在vm上
                //函数都写成普通函数，this表示vue实例，不要写成箭头函数，箭头函数中的this是window
                scroll(){
                    console.log('scroll滚动')
                },
                wheel(){
                    console.log('wheel')
                    for(let i=0;i<10000;i++){
                        console.log ('djfh')
                    }
                    console.log('累死')
                }
            }
     })
</script>

<div id="root">

            },
             computed:{

                }
            },
          })

<div id="root">
    <h2>今天天气{{info}}，是的加{{a}}</h2>
    第二种方法，回调函数
    <button @click="clickweather">切换天气</button>
</div>

<script type="text/javascript">
        Vue.config.productionTip=false;
        const vm=new Vue ({
            el:'#root',
            data:{
                name:'vue6监听属性案例',
                weather:true,
                a:0
            },
            methods: {
                clickweather(){
                    this.weather=!this.weather,
                    this.a++
                }
            },
            computed:{
                info(){
                    return this.weather ? '炎热':'凉快'
                }
            },
            // 监视（在控制台）
            //监视对象内部值的改变（一层）
            watch:{
                // 监视谁
                weather:{
                    immediate:true,//初始化时让handle调用一下
                    //当weather发生改变时调用handle
                    handler(newValue,oldValue){
                        console.log('weather被修改，handler被调用',newValue,oldValue)
                    }
                },
                info:{
                    immediate:true,//初始化时让handle调用一下
                    //当weather发生改变时调用handle
                    handler(newValue,oldValue){
                        console.log('info被修改，handler被调用',newValue,oldValue)
                        console.log('调用前的值',newValue)
                        console.log('调用后的值',oldValue)
                    }
                }
           }
})

也可通过vm进行监视
vm.$watch('weather',{
     immediate:true,//初始化时让handle调用一下
     //当weather发生改变时调用handle
     handler(newValue,oldValue){
         console.log('通过vm实现监视')
         console.log('weather被修改，handler被调用',newValue,oldValue)
         console.log('调用前的值',newValue)
         console.log('调用后的值',oldValue)
      }
})

            },
            watch:{
                // 监视多级结构中某个属性的变化
                'numbers.b':{
                    handler(){
                        console.log('numbers中b的值被修改')
                    }
                },
                // 监视多级属性中所有属性的变化
                numbers:{
                    deep:true,//可以检测到对象内部值改变（多层）
                    handler(){
                        console.log('深度监视numbers中某值被修改')
                    }
                }
            }
})

深度监视是监视多级属性中所有属性的变化，只要有一个修改，就会被检测到

# 12，计算属性与监听属性的区别

监听属性代码是命令式且重复的

computed的能完成的功能，watch都可以完成，

watch能完成的功能，computed不一定能完成，例如watch可以开启异步任务定时操作

所有被vue管理的函数最好写成普通函数，这样this的指向才是vm或组件实例对象

所有不被vue管理的函数（定时器的回调函数，Ajax的回调函数，promise的回调函数），最好写成箭头函数，这样this的指向才是vm或组件实例对象

<div id="root">
        姓：<input type="text" v-model="firstname"><br>
        名：<input type="text" v-model="lastname"><br>
        全名：<span>{{fullname}}</span>

</div>
<script type="text/javascript">
        Vue.config.productionTip = false;
        const vm=new Vue({
            el:"#root",
            data:{
                firstname:'Su',
                lastname:'ting',
                fullname:'Su-ting'
            },
            // computed与watch配置项不冲突
            computed:{

            },
            watch:{
                firstname(newValue){
                    // 等一秒在执行
                    setTimeout(() => {
                        this.fullname=newValue+'-'+this.lastname
                    }, 1000);
                },
                lastname(newValue){
                    this.fullname=this.firstname+'-'+newValue
                }
            }
        })
</script>
